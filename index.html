// 通用工具函数 const Utils = { // 格式化时间 formatTime(date) { return date.toLocaleString('zh-CN'); }, // 生成随机延迟 randomDelay(min, max) { return Math.floor(Math.random() * (max - min + 1)) + min; }, // 数据验证 validatePhone(phone) { return /^\+?[\d\s\-\(\)]+$/.test(phone); }, // 导出数据 exportData(data, filename) { const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = filename; a.click(); URL.revokeObjectURL(url); }, // 导入数据 importData(file) { return new Promise((resolve, reject) => { const reader = new FileReader(); reader.onload = (e) => { try { resolve(JSON.parse(e.target.result)); } catch (error) { reject(error); } }; reader.readAsText(file); }); } };
